const API = "https://pickit-bskz.onrender.com"; let user = null; try { user = JSON.parse(localStorage.getItem("user")); } catch (e) { user = null; } if (!user || !user.phone) { window.location.href = "index.html"; } async function refreshBalance() { try { const res = await fetch(API + "/balance", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ phone: user.phone }) }); const data = await res.json(); document.getElementById("balance").innerText = "Balance: $" + data.balance; } catch (err) { console.log("Balance error", err); } } async function place() { try { await fetch(API + "/place", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ phone: user.phone, choice: "black" }) }); alert("Placed $0.20 game"); refreshBalance(); } catch (e) { alert("Error placing game"); } } async function pick() { const gameId = prompt("Enter Game ID"); try { await fetch(API + "/pick", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ phone: user.phone, gameId: gameId, choice: "white" }) }); alert("Game played"); refreshBalance(); } catch (e) { alert("Error playing game"); } } function logout() { localStorage.removeItem("user"); window.location.href = "index.html"; } refreshBalance();
